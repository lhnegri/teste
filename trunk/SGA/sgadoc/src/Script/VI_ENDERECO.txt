CREATE VIEW VI_ENDERECO(
    CODIGO,
    CODIGO_CIDADE,
    TIPO_LOGRADOURO,
    LOGRADOURO,
    COMPLEMENTO,
    TIPO_LOGRADOURO_COMPLEMENTO,
    CEP,
    BAIRRO,
    CIDADE,
    UF)
AS
SELECT LO.CODIGO,
       LO.CODIGO_CIDADE,
       TI.NOME,
       TI.ABREVIACAO || ' ' || LO.NOME,
       LO.COMPLEMENTO,
       TI.ABREVIACAO || ' ' || LO.NOME || ' ' || LO.COMPLEMENTO,
       LO.CEP,
       BA.NOME_EXTENSO,
       CI.NOME,
       UF.SIGLA
FROM END_LOGRADOURO LO
JOIN END_CIDADE CI ON (CI.CODIGO = LO.CODIGO_CIDADE)
JOIN END_UF UF ON (UF.CODIGO = CI.CODIGO_UF)
JOIN END_TIPO TI ON (TI.CODIGO = LO.CODIGO_TIPO)
JOIN END_BAIRRO BA ON (BA.CODIGO = LO.CODIGO_BAIRRO AND BA.CODIGO_CIDADE = LO.CODIGO_CIDADE)
;
;
