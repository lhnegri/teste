CREATE TABLE PRODUTO (
    ID                  INTEGER NOT NULL,
    DESCRICAO           VARCHAR(200) NOT NULL,
    VALOR_COMPRA        DECIMAL(15,2) NOT NULL,
    CODIGO_FABRICANTE   VARCHAR(40) NOT NULL,
    ESTOQUE             INTEGER NOT NULL,
    MARGEM_LUCRO        DECIMAL(15,2) NOT NULL,
    LOCALIZACAO         VARCHAR(30) NOT NULL,
    VALOR_VENDA         DECIMAL(15,2) NOT NULL,
    ID_GRUPO            INTEGER NOT NULL,
    QTD_MINIMA          INTEGER NOT NULL,
    IMPOSTO             DECIMAL(15,2) NOT NULL,
    DATA_ATUALIZACAO    DATE,
    OBSERVACAO          VARCHAR(500),
    QTD_MAXIMA          INTEGER NOT NULL,
    DATA_ULTIMA_COMPRA  DATE,
    PRECO_PROMOCIONAL   DECIMAL(15,2),
    UNIDADE             VARCHAR(80)
);

ALTER TABLE PRODUTO ADD CONSTRAINT PK_PRODUTO PRIMARY KEY (ID);

ALTER TABLE PRODUTO ADD CONSTRAINT FK_PRODUTO_2 FOREIGN KEY (ID_GRUPO) REFERENCES GRUPO (ID) ON DELETE CASCADE ON UPDATE CASCADE;
