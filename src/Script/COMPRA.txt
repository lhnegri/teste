    ID_FORNECEDOR       INTEGER NOT NULL,
    HORA_COMPRA         TIME NOT NULL,
    NUMERO_DOCUMENTO    INTEGER NOT NULL,
    STATUS              VARCHAR(5) NOT NULL
);
