CREATE TABLE END_LOGRADOURO (
    CODIGO         INTEGER NOT NULL,
    NOME           VARCHAR(60),
    CODIGO_CIDADE  INTEGER NOT NULL,
    CODIGO_BAIRRO  INTEGER,
    CEP            VARCHAR(8),
    UF             VARCHAR(2),
    CODIGO_TIPO    INTEGER,
    COMPLEMENTO    VARCHAR(65)
);

ALTER TABLE END_LOGRADOURO ADD CONSTRAINT PK_END_LOGRADOURO PRIMARY KEY (CODIGO, CODIGO_CIDADE);


ALTER TABLE END_LOGRADOURO ADD CONSTRAINT FK_END_LOGRADOURO_BAIRRO FOREIGN KEY (CODIGO_BAIRRO, CODIGO_CIDADE) REFERENCES END_BAIRRO (CODIGO, CODIGO_CIDADE);
ALTER TABLE END_LOGRADOURO ADD CONSTRAINT FK_END_LOGRADOURO_CIDADE FOREIGN KEY (CODIGO_CIDADE) REFERENCES END_CIDADE (CODIGO);
ALTER TABLE END_LOGRADOURO ADD CONSTRAINT FK_END_LOGRADOURO_TIPO FOREIGN KEY (CODIGO_TIPO) REFERENCES END_TIPO (CODIGO);
